# bashrc controls
TEMP_BASHRC_PATH=/home/stvfu/script
alias bbb="sudo vi $TEMP_BASHRC_PATH/.bashrc"
alias srcbbb=save_bashrc

save_bashrc()
{
    sudo cp $TEMP_BASHRC_PATH/.bashrc.1 $TEMP_BASHRC_PATH/.bashrc.2
    sudo cp ~/.bashrc $TEMP_BASHRC_PATH/.bashrc.1

    sudo cp $TEMP_BASHRC_PATH/.bashrc ~/.bashrc
    source ~/.bashrc
}

# folder change
alias z="cd_upper_layer"
alias x="cd_down_layer"
alias cdh="cd /home/stvfu"

# cd upper layer
cd_upper_layer()
{
    pushd . > /dev/null
    cd ..
    pwd_path=`pwd`
    echo -e "\e[0m${pwd_path}\e[0m"
    ls
}
# cd last back layer
cd_down_layer()
{
    popd > /dev/null
    # dirname_path=`dirname "$PWD"`
    # basename_path=`basename "$PWD"`
    pwd_path=`pwd`
    echo -e "\e[0m${pwd_path}\e[0m"
    ls
}

# find / grep 
alias=grep_i
grep_i()
{
    grep -Iirn $1
}
alias fn="find_1"
find_1()
{
    find . -name "$1*" | grep $1
}

alias dss="folder_size"
folder_size()
{
    du -ch --max-depth=1
}
